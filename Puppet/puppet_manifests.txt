  ensure => installed,
}

# install wget package
package { 'wget':
  require => Exec['yum-update'],        # require 'yum-update' before installing
  ensure => installed,
}

# install httpd package
package { 'httpd':
  require => Exec['yum-update'],        # require 'yum-update' before installing
  ensure => installed,
}

# ensure httpd service is running
service { 'httpd':
  ensure => running,
}
